## ข้อแปลกของ document.cookie

`document.cookie` ไม่ทำงานเหมือน property JavaScript ทั่วไป มันมีพฤติกรรมที่แตกต่างกันโดยสิ้นเชิงขึ้นอยู่กับว่าคุณกำลังอ่านหรือเขียน

**การอ่าน** จะคืนคุกกี้ทั้งหมดสำหรับหน้าปัจจุบันเป็นสตริงเดียวคั่นด้วยเซมิโคลอน:

```js
// ถ้าตั้งค่าคุกกี้สามชิ้น:
console.log(document.cookie);
// "theme=dark; user=Alice; lang=en"
```

**การเขียน** จะตั้งค่า (หรืออัปเดต) คุกกี้เพียงหนึ่งชิ้น — ไม่ได้แทนที่คุกกี้ทั้งหมด:

```js
document.cookie = 'theme=light; Max-Age=86400; SameSite=Lax';
// อัปเดตเฉพาะ theme — คุกกี้อื่นไม่เปลี่ยนแปลง
```

ความไม่สมมาตรนี้คือส่วนที่น่าประหลาดใจที่สุดของ cookies API การเขียนไปยัง `document.cookie` เหมือนกับการเรียกฟังก์ชัน `setCookie()` มากกว่าการกำหนดค่าให้ property สตริง

## การตั้งค่าคุกกี้

ไวยากรณ์สำหรับตั้งค่าคุกกี้คือสตริงที่เริ่มด้วย name=value ตามด้วย attribute ที่เป็นตัวเลือก คั่นด้วยเซมิโคลอน:

```js
// Session cookie (ไม่มี Expires/Max-Age — ลบเมื่อเบราว์เซอร์ปิด)
document.cookie = 'demo_pref=dark; SameSite=Lax';

// Persistent cookie (รอดหลังเบราว์เซอร์ restart)
document.cookie = 'demo_pref=dark; Max-Age=86400; SameSite=Lax';

// Cookie ที่จำกัด path
document.cookie = 'demo_pref=dark; Max-Age=86400; Path=/app; SameSite=Lax';
```

ควรใส่ `SameSite=Lax` เป็นอย่างน้อยสำหรับคุกกี้ที่ตั้งฝั่ง client เสมอ เบราว์เซอร์สมัยใหม่ default เป็น `Lax` สำหรับคุกกี้ที่ไม่มี `SameSite` แต่การระบุชัดเจนปลอดภัยกว่า

## การอ่านคุกกี้ตามชื่อ

เนื่องจากการอ่าน `document.cookie` คืนคุกกี้ทั้งหมดเป็นสตริงเดียว คุณต้อง parse เองเพื่อหาคุกกี้ที่ต้องการตามชื่อ:

```js
function getCookie(name) {
  var pairs = document.cookie.split('; ');
  for (var i = 0; i < pairs.length; i++) {
    var pair = pairs[i].split('=');
    if (pair[0] === name) return decodeURIComponent(pair[1]);
  }
  return null;
}

var theme = getCookie('demo_pref'); // 'dark' หรือ null
```

สังเกต `decodeURIComponent` — ค่าที่ตั้งด้วย `encodeURIComponent` ต้องถูก decode ตอนอ่านด้วย

## การลบคุกกี้

ไม่สามารถลบคุกกี้ได้ด้วยการกำหนดสตริงว่าง การลบคุกกี้ทำได้โดยการตั้งค่าใหม่พร้อม `Max-Age=0` (หรือ `Expires` ที่ผ่านมาแล้ว):

```js
function deleteCookie(name) {
  document.cookie = name + '=; Max-Age=0; SameSite=Lax';
}

deleteCookie('demo_pref');
```

## รูปแบบ helper ที่ครบชุด

นี่คือรูปแบบ get/set/delete ที่ใช้ในโค้ด production:

```js
function setCookie(name, value, maxAge) {
  document.cookie = name + '=' + encodeURIComponent(value) +
    '; Max-Age=' + maxAge + '; SameSite=Lax';
}

function getCookie(name) {
  var pairs = document.cookie.split('; ');
  for (var i = 0; i < pairs.length; i++) {
    var pair = pairs[i].split('=');
    if (pair[0] === name) return decodeURIComponent(pair[1]);
  }
  return null;
}

function deleteCookie(name) {
  document.cookie = name + '=; Max-Age=0; SameSite=Lax';
}
```
